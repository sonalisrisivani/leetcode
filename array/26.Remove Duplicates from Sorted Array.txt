class Solution {
    public int removeDuplicates(int[] nums) {
        if (nums.length == 0) {
            return 0;
        }

        int j = 1; // Index for placing the next unique element.
        
        // Iterate from the second element to the end.
        for (int i = 1; i < nums.length; i++) {
            if (nums[i] != nums[i - 1]) {
                nums[j] = nums[i]; // Place the unique element.
                j++;
            }
        }
        
        return j; // Return the length of the array with unique elements.
    }
}
